  build:

    steps:
      - name: Checkout
        uses: actions/checkout@v4
      - name: Setup Pages
        uses: actions/configure-pages@v4

        uses: actions/upload-pages-artifact@v3
        with:
          path: './public'
      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4

function escapeHtml(str) {
  if (!str) return '';
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&#39;');
}

// Wait for auth to be ready
firebase.auth().onAuthStateChanged(async (user) => {
  console.log("Auth state changed:", user ? user.email : "No user");
  
  if (!user) {
    document.getElementById('container').innerHTML = `
      <div class="profile-card">
        <p>Please sign in to view your account.</p>
        <a href="../login/index.html" style="display:inline-block;margin-top:16px; color:var(--accent); text-decoration:none;">Sign In →</a>
      </div>
    `;
    return;
  }

  try {
    // Load user's mods from 'mods' path (NOT 'mods-list')
    const modsSnap = await firebase.database().ref('mods').orderByChild('authorId').equalTo(user.uid).once('value');
    console.log("Mods snapshot exists:", modsSnap.exists());
    
    let myMods = [];
    if (modsSnap.exists()) {
      modsSnap.forEach(child => {
        myMods.push({ id: child.key, ...child.val() });
      });
    }
    myMods.reverse(); // newest first

    let modsHtml = '';
    if (myMods.length === 0) {
      modsHtml = '<div class="empty-msg">You haven\'t published any mods yet.<br><a href="../mods/index.html" style="color:var(--accent);">Create your first mod →</a></div>';
    } else {
      modsHtml = '<div class="mods-grid">';
      myMods.forEach(mod => {
        modsHtml += `
          <a href="../view-mod/index.html?id=${mod.id}" class="mod-card">
            <div class="mod-card-title">${escapeHtml(mod.modName)}</div>
            <div class="mod-card-game">${escapeHtml(mod.gameName)}</div>
          </a>
        `;
      });
      modsHtml += '</div>';
    }

    // Render the page
    document.getElementById('container').innerHTML = `
      <div class="profile-card">
        <div class="avatar">${escapeHtml((user.displayName || user.email || 'U')[0].toUpperCase())}</div>
        <h2>${escapeHtml(user.displayName || 'Modder')}</h2>
        <p style="color:var(--muted); font-size:13px;">${escapeHtml(user.email || '')}</p>
        <button class="logout-btn" id="logoutBtn" style="margin-top:20px;">Sign Out</button>
      </div>
      <div class="section">
        <div class="section-title">📦 My Mods (${myMods.length})</div>
        ${modsHtml}
      </div>
    `;

    // Add logout handler
    const logoutBtn = document.getElementById('logoutBtn');
    if (logoutBtn) {
      logoutBtn.addEventListener('click', () => {
        firebase.auth().signOut();
        window.location.href = '../main/index.html';
      });
    }
    
  } catch (error) {
    console.error("Error loading account:", error);
    document.getElementById('container').innerHTML = `
      <div class="profile-card">
        <p style="color: #ff6b60;">Error loading account data.</p>
        <p style="font-size: 12px; margin-top: 8px;">${escapeHtml(error.message)}</p>
        <a href="../main/index.html" style="display:inline-block;margin-top:16px; color:var(--accent);">Go Back →</a>
      </div>
    `;
  }
});

async function submitMod() {
  if (!document.getElementById('confirmCheck').checked) {
    alert('Please confirm you own this mod before publishing.');
    return;
  }

  const user = firebase.auth().currentUser;
  if (!user) {
    alert('Session expired. Please sign in again.');
    window.location.href = '../login/index.html';
    return;
  }

  const name = document.getElementById('modName').value.trim();
  const desc = document.getElementById('modDesc').value.trim();
  const game = document.getElementById('modGame').value;
  const fileUrl = document.getElementById('modFileUrl').value.trim();
  const installDesc = document.getElementById('installDesc').value.trim();
  const imageUrlsRaw = document.getElementById('imageUrls').value;
  const imageUrls = imageUrlsRaw.split('\n').filter(u => u.trim() !== '');

  if (!name || !desc || !game || !fileUrl || imageUrls.length === 0) {
    alert('Missing required fields. Please go back and check.');
    return;
  }

  if (!isValidUrl(fileUrl)) { alert('The download link is not a valid URL.'); return; }
  for (let i = 0; i < imageUrls.length; i++) {
    if (!isValidUrl(imageUrls[i].trim())) { alert(`Image URL #${i+1} is not valid.`); return; }
  }

  const publishBtn = document.getElementById('publishBtn');
  publishBtn.disabled = true;
  publishBtn.innerHTML = '<span class="spinner"></span>Publishing...';

  try {
    const newRef = firebase.database().ref('mods').push();
    await newRef.set({
      modName: name,
      modDesc: desc,
      gameName: game,
      installInstructions: installDesc,
      modFileUrl: fileUrl,
      mediaUrls: imageUrls,
      authorId: user.uid,
      authorName: user.displayName || user.email || 'Anonymous',
      authorEmail: user.email,
      createdAt: Date.now()
    });
    window.location.href = `../view-mod/index.html?id=${newRef.key}`;
  } catch (err) {
    console.error(err);
    alert('Error publishing mod: ' + err.message);
    publishBtn.disabled = false;
    publishBtn.innerHTML = 'Publish Mod';
  }
}

function renderMod(mod) {

async function deleteMod() {
  const user = firebase.auth().currentUser;
  const params = new URLSearchParams(window.location.search);
  const modId = params.get('id');
  
  if (!user) { 
    alert('Please sign in first.'); 
    return; 
  }
  
  try {
    // التحقق من ملكية المود قبل الحذف
    const snap = await firebase.database().ref('mods/' + modId).once('value');
    if (!snap.exists()) { 
      alert('Mod not found.'); 
      return; 
    }
    if (snap.val().authorId !== user.uid) { 
      alert('You can only delete your own mods.'); 
      return; 
    }
    
    if (!confirm('⚠️ Are you sure you want to permanently delete this mod? This cannot be undone.')) return;
    
    await firebase.database().ref('mods/' + modId).remove();
    alert('✅ Mod deleted successfully.');
    window.location.href = '../main/index.html';
