      .replace(/"/g,'&quot;')
      .replace(/'/g,'&#39;');

  firebase.database().ref('mods-list/' + modId).once('value').then(snap => {
    if (!snap.exists()) {
      document.getElementById('content').innerHTML = `
        <div class="state-center"><div class="icon">🔍</div><p>Mod not found.</p></div>`;
      return;
    }

    const mod = snap.val();
    renderMod(mod);

    // Check auth for delete button
    firebase.auth().onAuthStateChanged(user => {
      if (user && user.uid === mod.authorId) {
        const delBtn = document.getElementById('delete-btn');
        if (delBtn) delBtn.style.display = 'inline-flex';
      }
    });
  }).catch(() => {
    document.getElementById('content').innerHTML = `
      <div class="state-center"><div class="icon">⚠️</div><p>Failed to load mod. Try again later.</p></div>`;
  });
}

function renderMod(mod) {
  const media  = mod.mediaUrls || [];
  const date   = mod.createdAt ? new Date(mod.createdAt).toLocaleDateString('en-GB', { year:'numeric', month:'long', day:'numeric' }) : 'Unknown date';

  // Build hero
  let heroContent = '';
  if (media.length > 0) {
    const isVideo = /\.(mp4|webm|mov|avi)$/i.test(media[0]) || media[0].includes('video');
    if (isVideo) {
      heroContent = `<video id="hero-media" src="${escapeHtml(media[0])}" controls muted loop playsinline style="width:100%;height:100%;object-fit:cover"></video>`;
    } else {
      heroContent = `<img id="hero-media" src="${escapeHtml(media[0])}" alt="${escapeHtml(mod.modName || '')}">`;
    }
  } else {
    heroContent = `<div class="media-hero-placeholder">🎮</div>`;
  }

  // Thumbnails
  let thumbsHtml = '';
  if (media.length > 1) {
    thumbsHtml = '<div class="thumb-row" id="thumb-row">';
    media.forEach((url, i) => {
      const isVid = /\.(mp4|webm|mov|avi)$/i.test(url) || url.includes('video');
      const inner = isVid
        ? `<video src="${escapeHtml(url)}" muted playsinline style="width:100%;height:100%;object-fit:cover;pointer-events:none"></video>`
        : `<img src="${escapeHtml(url)}" alt="Screenshot ${i+1}">`;
      thumbsHtml += `<div class="thumb${i===0?' active':''}" data-index="${i}" data-url="${escapeHtml(url)}" data-type="${isVid?'video':'img'}">${inner}</div>`;
    });
    thumbsHtml += '</div>';
  }

  // Download link
  const dlBtn = mod.modFileUrl
    ? `<a class="download-btn" href="${escapeHtml(mod.modFileUrl)}" target="_blank" rel="noopener noreferrer">
         <svg width="18" height="18" fill="none" stroke="currentColor" stroke-width="2" viewBox="0 0 24 24"><path d="M21 15v4a2 2 0 01-2 2H5a2 2 0 01-2-2v-4"/><polyline points="7 10 12 15 17 10"/><line x1="12" y1="15" x2="12" y2="3"/></svg>
         Download Mod
       </a>`
    : `<span style="color:var(--muted);font-size:14px">No download link provided</span>`;

  // Install instructions
  const installHtml = mod.installInstructions
    ? `<div class="section">
         <div class="section-title">Installation Instructions</div>
         <p>${escapeHtml(mod.installInstructions)}</p>
       </div>`
    : '';

  document.getElementById('content').innerHTML = `
    <div class="media-hero">${heroContent}</div>
    ${thumbsHtml}

    <div class="mod-header">
      <div>
        <div class="mod-game-tag">${escapeHtml(mod.gameName || 'Unknown Game')}</div>
        <div class="mod-title">${escapeHtml(mod.modName || 'Untitled Mod')}</div>
        <div class="mod-meta">
          <span>By ${escapeHtml(mod.authorName || 'Anonymous')}</span>
          <span>Published ${date}</span>
        </div>
      </div>
      ${dlBtn}
    </div>

    <div class="section">
      <div class="section-title">About This Mod</div>
      <p>${escapeHtml(mod.modDesc || 'No description provided.')}</p>
    </div>

    ${installHtml}

    <button class="danger-btn" id="delete-btn" style="display:none" onclick="deleteMod()">
      <svg width="16" height="16" fill="none" stroke="currentColor" stroke-width="2" viewBox="0 0 24 24"><polyline points="3 6 5 6 21 6"/><path d="M19 6l-1 14a2 2 0 01-2 2H8a2 2 0 01-2-2L5 6"/><path d="M10 11v6M14 11v6"/><path d="M9 6V4a1 1 0 011-1h4a1 1 0 011 1v2"/></svg>
      Delete This Mod
    </button>
  `;

  // ── Thumbnail switcher
  const thumbRow = document.getElementById('thumb-row');
  if (thumbRow) {
    thumbRow.addEventListener('click', e => {
      const thumb = e.target.closest('.thumb');
      if (!thumb) return;

      thumbRow.querySelectorAll('.thumb').forEach(t => t.classList.remove('active'));
      thumb.classList.add('active');

      const heroWrap = document.querySelector('.media-hero');
      const url  = thumb.dataset.url;
      const type = thumb.dataset.type;

      if (type === 'video') {
        heroWrap.innerHTML = `<video src="${url}" controls muted loop playsinline style="width:100%;height:100%;object-fit:cover"></video>`;
      } else {
        heroWrap.innerHTML = `<img src="${url}" alt="Screenshot" style="width:100%;height:100%;object-fit:cover">`;
      }
    });
  }
}

async function deleteMod() {
  const user = firebase.auth().currentUser;
  if (!user) return;

  const params = new URLSearchParams(window.location.search);
  const modId = params.get('id');

  if (!confirm('Are you sure you want to permanently delete this mod? This cannot be undone.')) return;

  try {
    await firebase.database().ref('mods-list/' + modId).remove();
    alert('Mod deleted successfully.');
    window.location.href = '../main/index.html';
  } catch (err) {
    alert('Error deleting mod: ' + err.message);
  }
}
